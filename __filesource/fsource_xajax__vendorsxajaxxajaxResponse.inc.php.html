  303.     /**
  304.      * Uses the addScript() method to add a Javascript redirect to another URL.

  310.     function addRedirect($sURL, $iDelay=0)
  311.     {
  312.         //we need to parse the query part so that the values are rawurlencode()'ed
  313.         //can't just use parse_url() cos we could be dealing with a relative URL which
  314.         //  parse_url() can't deal with.
  315.         $queryStart = strpos($sURL, '?', strrpos($sURL, '/'));
  316.         if ($queryStart !== FALSE)
  317.         {
  318.             $queryStart++;
  319.             $queryEnd = strpos($sURL, '#', $queryStart);
  320.             if ($queryEnd === FALSE)
  321.                 $queryEnd = strlen($sURL);
  322.             $queryPart = substr($sURL, $queryStart, $queryEnd-$queryStart);
  323.             parse_str($queryPart, $queryParts);
  324.             $newQueryPart = "";
  325.             if ($queryParts)
  326.             {
  327.                 $first = true;
  328.                 foreach($queryParts as $key => $value)
  329.                 {
  330.                     if ($first)
  331.                         $first = false;
  332.                     else
  333.                         $newQueryPart .= ini_get('arg_separator.output');
  334.                     $newQueryPart .= rawurlencode($key).'='.rawurlencode($value);
  335.                 }
  336.             } else if ($_SERVER['QUERY_STRING']) {
  337.                 //couldn't break up the query, but there's one there
  338.                 //possibly "http://url/page.html?query1234" type of query?
  339.                 //just encode it and hope it works
  340.                 $newQueryPart = rawurlencode($_SERVER['QUERY_STRING']);
  341.             }
  342.             $sURL = str_replace($queryPart, $newQueryPart, $sURL);
  343.         }
  344.         if ($iDelay)
  345.             $this->addScript('window.setTimeout("window.location = \''.$sURL.'\';",'.($iDelay*1000).');');
  346.         else
  347.             $this->addScript('window.location = "'.$sURL.'";');
  348.         return $this;
  349.     }

  755.     /**

  760.      * @return string XML command
  761.      */
  762.     function _arrayToXML($mArray) {
  763.         if (!is_array($mArray))
  764.             return $this->_escape($mArray);
  765.         
  766.         $xml = '<xjxobj>';
  767.         foreach ($mArray as $aKey=>$aKeyValues) {

  773.                     $xml .= '</'.htmlentities($sKey).'>';    
  774.                 }

  776.             } else {

  779.                 $xml .= '</k><v>';
  780.                 $xml .= $this->_escape($aKeyValues);
  781.                 $xml .= '</v></e>';
  782.             }
  783.         }

  801.     /**
  802.      * Escapes the data.  Can be overridden to allow other transports to send
  803.      * data.
  804.      * 
  805.      * @access private
  806.      * @param string data
  807.      * @return string escaped data
  808.      */
  809.     function _escape($sData)
  810.     {
  811.         if (!is_numeric($sData) && !$sData)
  812.             return '';
  813.         
  814.         $needCDATA = false;
  815.         
  816.         if ($this->bOutputEntities) {
  817.             if (!function_exists('mb_convert_encoding'))
  818.                 trigger_error('The xajax response output could not be converted to HTML entities because the mb_convert_encoding function is not available', E_USER_NOTICE);
  819.             
  820.             $sData = call_user_func_array('mb_convert_encoding', array(&$sData, 'HTML-ENTITIES', $this->sEncoding));
  821.         } else {
  822.             if ((strpos($sData, '<![CDATA[') !== false)
  823.             || (strpos($sData, ']]>') !== false)
  824.             || (htmlentities($sData) != $sData))
  825.                 $needCDATA = true;
  826.             
  827.             $segments = explode('<![CDATA[', $sData);
  828.             $sData = '';
  829.             foreach ($segments as $key => $segment) {
  830.                 $fragments = explode(']]>', $segment);
  831.                 $segment = '';
  832.                 foreach ($fragments as $fragment) {
  833.                     if ($segment != '')
  834.                         $segment .= ']]]]><![CDATA[>';
  835.                     $segment .= $fragment;
  836.                 }
  837.                 if ($sData != '')
  838.                     $sData .= '<![]]><![CDATA[CDATA[';
  839.                 $sData .= $segment;
  840.             }
  841.         }
  842.         
  843.         if ($needCDATA)
  844.             $sData = '<![CDATA['.$sData.']]>';
  845.         
  846.         return $sData;
  847.     }

  849.     /**
  850.      * Recursively serializes a data structure in an array so it can be sent to
