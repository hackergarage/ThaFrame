  84.   /**
  85.    * An ls style command using regular expresions

  87.    * By fordiman@gmail.com taken from PHP documentation:
  88.    * http://www.php.net/manual/en/class.dir.php#60562

  92.    * @param boolean $recursive if the ls should be recursive
  93.    * @param string $patttern the pattern in regular expression format
  94.    * @return Array 
  95.    */ 
  96.   public static function preg_ls ($path=".", $recursive=false, $pattern="/.*/") {
  97.     $rec = $recursive;
  98.     $pat = $pattern;
  99.     // it's going to be used repeatedly, ensure we compile it for speed.
  100.     $pat=preg_replace("|(/.*/[^S]*)|s", "\\1S", $pat);
  101.     //Remove trailing slashes from path
  102.     while (substr($path,-1,1)=="/") $path=substr($path,0,-1);
  103.     //also, make sure that $path is a directory and repair any screwups
  104.     if (!is_dir($path)) $path=dirname($path);
  105.     //assert either truth or falsehoold of $rec, allow no scalars to mean truth
  106.     if ($rec!==true) $rec=false;
  107.     //get a directory handle
  108.     $d=dir($path);
  109.     //initialise the output array
  110.     $ret=Array();
  111.     //loop, reading until there's no more to read
  112.     while (false!==($e=$d->read())) {
  113.         //Ignore parent- and self-links
  114.         if (($e==".")||($e=="..")) continue;
  115.         //If we're working recursively and it's a directory, grab and merge
  116.         if ($rec && is_dir($path."/".$e)) {
  117.             $ret=array_merge($ret,preg_ls($path."/".$e,$rec,$pat));
  118.             continue;
  119.         }
  120.         //If it don't match, exclude it
  121.         if (!preg_match($pat,$e)) continue;
  122.         //In all other cases, add it to the output array
  123.         $ret[]=$path."/".$e;
  124.     }
  125.     //finally, return the array
  126.     return $ret;
  127.   }
