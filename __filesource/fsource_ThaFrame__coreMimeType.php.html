  36. /**
  37.    Name: PHP MimeType Class
  38.  
  39.    Version: 1.0
  40.    Date Released: 10/20/02
  41.  
  42.    Description: This class allows a PHP script to determine the mime type
  43.    a file based on the file extension.  This class is handy for PHP versions
  44.    without the benefit of other tools to determine the mime type.  The class
  45.    defaults to octet-stream so if the file type is not recognized the user
  46.    is presented with a file download prompt.
  47.  
  48.    NOTES: The mime types for this version are based on Apache 1.3.27
  49.    mime types may change or need to be added in the future, the mime types
  50.    are stored in an array so that an awk or other script can automatically
  51.    generate the code to make the array.
  52.  
  53.    Usage:
  54.  
  55.     First an instance of the mimetype class must be created, then the
  56.   getType method should be called with the filename.  It will return
  57.   the mime type, an example follows.
  58.   
  59.    Example:
  60.  
  61.       $mimetype = new mimetype();
  62.       print $mimetype->getType('acrobat.pdf');
  63.  
  64.    Author: Jason Sheets <jsheets@shadonet.com>
  65.  
  66.    License: This script is distributed under the BSD License, you are free
  67.    to use, or modify it however you like.  If you find this script useful please

  72. class MimeType {
  73.   /**
  74.    * Gets the mimetype string based in the filename extension
  75.    *
  76.    * @param string $filename 
  77.    * @return string 
  78.    */
  79.   public function getType($filename) {
  80.       // get base name of the filename provided by user
  81.       $filename = basename($filename);
  82.  
  83.       // break file into parts seperated by .
  84.       $filename = explode('.', $filename);
  85.  
  86.       // take the last part of the file to get the file extension
  87.       $filename = $filename[count($filename)-1];
  88.  
  89.       // find mime type
  90.       return $this->privFindType($filename);
  91.    }
  92.  
  93.    private function privFindType($ext) {
  94.       // create mimetypes array
  95.       $mimetypes = $this->privBuildMimeArray();
  96.       
  97.       // return mime type for extension
  98.       if (isset($mimetypes[$ext])) {
  99.          return $mimetypes[$ext];
  100.       // if the extension wasn't found return octet-stream
  101.       } else {
  102.          return 'application/octet-stream';
  103.       }
  104.          
  105.    }
