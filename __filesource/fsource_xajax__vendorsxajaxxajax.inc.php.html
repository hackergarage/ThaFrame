  478.      * array like so:
  479.      * <kbd>array("myFunctionName", "myClass", "myMethod")</kbd>
  480.      * For an object instance method, use an object variable for the second
  481.      * array element (and in PHP 4 make sure you put an & before the variable
  482.      * to pass the object by reference). Note: the function name is what you
  483.      * call via Javascript, so it can be anything as long as it doesn't
  484.      * conflict with any other registered function name.
  485.      * 
  486.      * <i>Usage:</i> <kbd>$xajax->registerFunction("myFunction");</kbd>
  487.      * or: <kbd>$xajax->registerFunction(array("myFunctionName", &$myObject, "myMethod"));</kbd>
  488.      * 
  489.      * @param mixed  contains the function name or an object callback array
  490.      * @param mixed  request type (XAJAX_GET/XAJAX_POST) that should be used
  491.      *                for this function.  Defaults to XAJAX_POST.
  492.      */
  493.     function registerFunction($mFunction,$sRequestType=XAJAX_POST)
  494.     {
  495.         if (is_string($sRequestType)) {
  496.             return $this->registerExternalFunction($mFunction, $sRequestType);
  497.         }
  498.     
  499.         if (is_array($mFunction)) {
  500.             $this->aFunctions[$mFunction[0]] = 1;
  501.             $this->aFunctionRequestTypes[$mFunction[0]] = $sRequestType;
  502.             $this->aObjects[$mFunction[0]] = array_slice($mFunction, 1);
  503.         }    
  504.         else {
  505.             $this->aFunctions[$mFunction] = 1;
  506.             $this->aFunctionRequestTypes[$mFunction] = $sRequestType;
  507.         }
  508.     }
  509.     
  510.     /**
  511.      * Registers a PHP function to be callable through xajax which is located
  512.      * in some other file.  If the function is requested the external file will
  513.      * be included to define the function before the function is called.
  514.      * 
  515.      * <i>Usage:</i> <kbd>$xajax->registerExternalFunction("myFunction","myFunction.inc.php",XAJAX_POST);</kbd>
  516.      * 
  517.      * @param string contains the function name or an object callback array
  518.      *                ({@link xajax::registerFunction() see registerFunction} for
  519.      *                more info on object callback arrays)
  520.      * @param string contains the path and filename of the include file
  521.      * @param mixed  the RequestType (XAJAX_GET/XAJAX_POST) that should be used
  522.      *                   for this function. Defaults to XAJAX_POST.
  523.      */
  524.     function registerExternalFunction($mFunction,$sIncludeFile,$sRequestType=XAJAX_POST)
  525.     {
  526.         $this->registerFunction($mFunction, $sRequestType);
  527.         
  528.         if (is_array($mFunction)) {
  529.             $this->aFunctionIncludeFiles[$mFunction[0]] = $sIncludeFile;
  530.         }
  531.         else {
  532.             $this->aFunctionIncludeFiles[$mFunction] = $sIncludeFile;
  533.         }
  534.     }
  535.     
  536.     /**
  537.      * Registers a PHP function to be called when xajax cannot find the
  538.      * function being called via Javascript. Because this is technically
  539.      * impossible when using "wrapped" functions, the catch-all feature is
  540.      * only useful when you're directly using the xajax.call() Javascript
  541.      * method. Use the catch-all feature when you want more dynamic ability to
  542.      * intercept unknown calls and handle them in a custom way.
  543.      * 
  544.      * <i>Usage:</i> <kbd>$xajax->registerCatchAllFunction("myCatchAllFunction");</kbd>
  545.      * 
  546.      * @param string contains the function name or an object callback array
  547.      *                ({@link xajax::registerFunction() see registerFunction} for
  548.      *                more info on object callback arrays)
  549.      */
  550.     function registerCatchAllFunction($mFunction)
  551.     {
  552.         if (is_array($mFunction)) {
  553.             $this->sCatchAllFunction = $mFunction[0];
  554.             $this->aObjects[$mFunction[0]] = array_slice($mFunction, 1);
  555.         }
  556.         else {
  557.             $this->sCatchAllFunction = $mFunction;
  558.         }
  559.     }
  560.     
  561.     /**
  562.      * Registers a PHP function to be called before xajax calls the requested
  563.      * function. xajax will automatically add the request function's response
  564.      * to the pre-function's response to create a single response. Another
  565.      * feature is the ability to return not just a response, but an array with
  566.      * the first element being false (a boolean) and the second being the
  567.      * response. In this case, the pre-function's response will be returned to
  568.      * the browser without xajax calling the requested function.
  569.      * 
  570.      * <i>Usage:</i> <kbd>$xajax->registerPreFunction("myPreFunction");</kbd>
  571.      * 
  572.      * @param string contains the function name or an object callback array
  573.      *                ({@link xajax::registerFunction() see registerFunction} for
  574.      *                more info on object callback arrays)
  575.      */
  576.     function registerPreFunction($mFunction)
  577.     {
  578.         if (is_array($mFunction)) {
  579.             $this->sPreFunction = $mFunction[0];
  580.             $this->aObjects[$mFunction[0]] = array_slice($mFunction, 1);
  581.         }
  582.         else {
  583.             $this->sPreFunction = $mFunction;
  584.         }
  585.     }
  586.     
  587.     /**
  588.      * Returns true if xajax can process the request, false if otherwise.
  589.      * You can use this to determine if xajax needs to process the request or
  590.      * not.
  591.      * 
  592.      * @return boolean 
  593.      */ 
  594.     function canProcessRequests()
  595.     {
  596.         if ($this->getRequestMode() != -1) return true;
  597.         return false;
  598.     }
  599.     
  600.     /**
  601.      * Returns the current request mode (XAJAX_GET or XAJAX_POST), or -1 if
  602.      * there is none.
  603.      * 
  604.      * @return mixed 
  605.      */
  606.     function getRequestMode()
  607.     {
  608.         if (!empty($_GET["xajax"]))
  609.             return XAJAX_GET;
  610.         
  611.         if (!empty($_POST["xajax"]))
  612.             return XAJAX_POST;
  613.             
  614.         return -1;
  615.     }
  616.     
  617.     function processRequest()
  618.     {
  619.         return $this->processRequests();
  620.     }
  621.     
  622.     /**
  623.      * This is the main communications engine of xajax. The engine handles all
  624.      * incoming xajax requests, calls the apporiate PHP functions (or
  625.      * class/object methods) and passes the XML responses back to the
  626.      * Javascript response handler. If your RequestURI is the same as your Web
  627.      * page then this function should be called before any headers or HTML has
  628.      * been sent.
  629.      */
  630.     function processRequests()
  631.     {    
  632.         
  633.         $requestMode = -1;
  634.         $sFunctionName = "";
  635.         $bFoundFunction = true;
  636.         $bFunctionIsCatchAll = false;
  637.         $sFunctionNameForSpecial = "";
  638.         $aArgs = array();
  639.         $sPreResponse = "";
  640.         $bEndRequest = false;
  641.         
  642.         $requestMode = $this->getRequestMode();
  643.         if ($requestMode == -1) return;
  644.     
  645.         if ($requestMode == XAJAX_POST)
  646.         {
  647.             $sFunctionName = $_POST["xajax"];
  648.             
  649.             if (!empty($_POST["xajaxargs"])) 
  650.                 $aArgs = $_POST["xajaxargs"];
  651.         }
  652.         else
  653.         {    
  654.             header ("Expires: Mon, 26 Jul 1997 05:00:00 GMT");
  655.             header ("Last-Modified: " . gmdate("D, d M Y H:i:s") . " GMT");
  656.             header ("Cache-Control: no-cache, must-revalidate");
  657.             header ("Pragma: no-cache");
  658.             
  659.             $sFunctionName = $_GET["xajax"];
  660.             
  661.             if (!empty($_GET["xajaxargs"])) 
  662.                 $aArgs = $_GET["xajaxargs"];
  663.         }
  664.         
  665.         // Use xajax error handler if necessary
  666.         if ($this->bErrorHandler) {
  667.             $GLOBALS['xajaxErrorHandlerText'] = "";
  668.             set_error_handler("xajaxErrorHandler");
  669.         }
  670.         
  671.         if ($this->sPreFunction) {
  672.             if (!$this->_isFunctionCallable($this->sPreFunction)) {
  673.                 $bFoundFunction = false;
  674.                 $oResponse = new xajaxResponse();
  675.                 $oResponse->addAlert("Unknown Pre-Function ". $this->sPreFunction);
  676.             }
  677.         }
  678.         //include any external dependencies associated with this function name
  679.         if (array_key_exists($sFunctionName,$this->aFunctionIncludeFiles))
  680.         {
  681.             ob_start();
  682.             include_once($this->aFunctionIncludeFiles[$sFunctionName]);
  683.             ob_end_clean();
  684.         }
  685.         
  686.         if ($bFoundFunction) {
  687.             $sFunctionNameForSpecial = $sFunctionName;
  688.             if (!array_key_exists($sFunctionName, $this->aFunctions))
  689.             {
  690.                 if ($this->sCatchAllFunction) {
  691.                     $sFunctionName = $this->sCatchAllFunction;
  692.                     $bFunctionIsCatchAll = true;
  693.                 }
  694.                 else {
  695.                     $bFoundFunction = false;
  696.                     $oResponse = new xajaxResponse();
  697.                     $oResponse->addAlert("Unknown Function $sFunctionName.");
  698.                 }
  699.             }
  700.         }
  701.         
  702.         if ($bFoundFunction)
  703.         {
  704.             for ($i = 0; $i < sizeof($aArgs); $i++)
  705.             {
  706.                 // If magic quotes is on, then we need to strip the slashes from the args
  707.                 if (get_magic_quotes_gpc() == 1 && is_string($aArgs[$i])) {
  708.                 
  709.                     $aArgs[$i] = stripslashes($aArgs[$i]);
  710.                 }
  711.                 if (stristr($aArgs[$i],"<xjxobj>") != false)
  712.                 {
  713.                     $aArgs[$i] = $this->_xmlToArray("xjxobj",$aArgs[$i]);    
  714.                 }
  715.                 else if (stristr($aArgs[$i],"<xjxquery>") != false)
  716.                 {
  717.                     $aArgs[$i] = $this->_xmlToArray("xjxquery",$aArgs[$i]);    
  718.                 }
  719.                 else if ($this->bDecodeUTF8Input)
  720.                 {
  721.                     $aArgs[$i] = $this->_decodeUTF8Data($aArgs[$i]);    
  722.                 }
  723.             }
  724.  
  725.             if ($this->sPreFunction) {
  726.                 $mPreResponse = $this->_callFunction($this->sPreFunction, array($sFunctionNameForSpecial, $aArgs));
  727.                 if (is_array($mPreResponse) && $mPreResponse[0] === false) {
  728.                     $bEndRequest = true;
  729.                     $sPreResponse = $mPreResponse[1];
  730.                 }
  731.                 else {
  732.                     $sPreResponse = $mPreResponse;
  733.                 }
  734.                 if ($bEndRequest) $oResponse = $sPreResponse;
  735.             }
  736.             
  737.             if (!$bEndRequest) {
  738.                 if (!$this->_isFunctionCallable($sFunctionName)) {
  739.                     $oResponse = new xajaxResponse();
  740.                     $oResponse->addAlert("The Registered Function $sFunctionName Could Not Be Found.");
  741.                 }
  742.                 else {
  743.                     if ($bFunctionIsCatchAll) {
  744.                         $aArgs = array($sFunctionNameForSpecial, $aArgs);
  745.                     }
  746.                     $oResponse = $this->_callFunction($sFunctionName, $aArgs);
  747.                 }
  748.                 if (is_string($sResponse)) {
  749.                     $oResponse = new xajaxResponse();
  750.                     $oResponse->addAlert("No XML Response Was Returned By Function $sFunctionName.\n\nOutput: ".$oResponse);
  751.                 }
  752.                 else if ($sPreResponse != "") {
  753.                     $oNewResponse = new xajaxResponse($this->sEncoding, $this->bOutputEntities);
  754.                     $oNewResponse->loadXML($sPreResponse);
  755.                     $oNewResponse->loadXML($oResponse);
  756.                     $oResponse = $sNewResponse;
  757.                 }
  758.             }
  759.         }
  760.         
  761.         $sContentHeader = "Content-type: text/xml;";
  762.         if ($this->sEncoding && strlen(trim($this->sEncoding)) > 0)
  763.             $sContentHeader .= " charset=".$this->sEncoding;
  764.         header($sContentHeader);
  765.         if ($this->bErrorHandler && !empty( $GLOBALS['xajaxErrorHandlerText'] )) {
  766.             $oErrorResponse = new xajaxResponse();
  767.             $oErrorResponse->addAlert("** PHP Error Messages: **" . $GLOBALS['xajaxErrorHandlerText']);
  768.             if ($this->sLogFile) {
  769.                 $fH = @fopen($this->sLogFile, "a");
  770.                 if (!$fH) {
  771.                     $oErrorResponse->addAlert("** Logging Error **\n\nxajax was unable to write to the error log file:\n" . $this->sLogFile);
  772.                 }
  773.                 else {
  774.                     fwrite($fH, "** xajax Error Log - " . strftime("%b %e %Y %I:%M:%S %p") . " **" . $GLOBALS['xajaxErrorHandlerText'] . "\n\n\n");
  775.                     fclose($fH);
  776.                 }
  777.             }
  778.  
  779.             $oErrorResponse->loadXML($oResponse);
  780.             $oResponse = $oErrorResponse;
  781.             
  782.         }
  783.         if ($this->bCleanBuffer) while (@ob_end_clean());
  784.         print $oResponse->getOutput();
  785.         if ($this->bErrorHandler) restore_error_handler();
  786.         
  787.         if ($this->bExitAllowed)
  788.             exit();
  789.     }
  790.  
  791.     /**            
  792.      * Prints the xajax Javascript header and wrapper code into your page by
  793.      * printing the output of the getJavascript() method. It should only be
  794.      * called between the <pre><head> </head></pre> tags in your HTML page.
  795.      * Remember, if you only want to obtain the result of this function, use
  796.      * {@link xajax::getJavascript()} instead.
  797.      * 
  798.      * <i>Usage:</i>
  799.      * <code>
  800.      *  <head>
  801.      *        ...
  802.      *        < ?php $xajax->printJavascript(); ? >
  803.      * </code>
  804.      * 
  805.      * @param string the relative address of the folder where xajax has been
  806.      *                installed. For instance, if your PHP file is
  807.      *                "http://www.myserver.com/myfolder/mypage.php"
  808.      *                and xajax was installed in
  809.      *                "http://www.myserver.com/anotherfolder", then $sJsURI
  810.      *                should be set to "../anotherfolder". Defaults to assuming
  811.      *                xajax is in the same folder as your PHP file.
  812.      * @param string the relative folder/file pair of the xajax Javascript
  813.      *                engine located within the xajax installation folder.
  814.      *                Defaults to xajax_js/xajax.js.
  815.      */
  816.     function printJavascript($sJsURI="", $sJsFile=NULL)
  817.     {
  818.         print $this->getJavascript($sJsURI, $sJsFile);
  819.     }
  820.     
  821.     /**
  822.      * Returns the xajax Javascript code that should be added to your HTML page
  823.      * between the <kbd><head> </head></kbd> tags.
  824.      * 
  825.      * <i>Usage:</i>
  826.      * <code>
  827.      *  < ?php $xajaxJSHead = $xajax->getJavascript(); ? >
  828.      *    <head>
  829.      *        ...
  830.      *        < ?php echo $xajaxJSHead; ? >
  831.      * </code>
  832.      * 
  833.      * @param string the relative address of the folder where xajax has been
  834.      *                installed. For instance, if your PHP file is
  835.      *                "http://www.myserver.com/myfolder/mypage.php"
  836.      *                and xajax was installed in
  837.      *                "http://www.myserver.com/anotherfolder", then $sJsURI
  838.      *                should be set to "../anotherfolder". Defaults to assuming
  839.      *                xajax is in the same folder as your PHP file.
  840.      * @param string the relative folder/file pair of the xajax Javascript
  841.      *                engine located within the xajax installation folder.
  842.      *                Defaults to xajax_js/xajax.js.
  843.      * @return string 
  844.      */
  845.     function getJavascript($sJsURI="", $sJsFile=NULL)
  846.     {    
  847.         $html = $this->getJavascriptConfig();
  848.         $html .= $this->getJavascriptInclude($sJsURI, $sJsFile);
  849.         
  850.         return $html;
  851.     }
  852.     
  853.     /**
  854.      * Returns a string containing inline Javascript that sets up the xajax
  855.      * runtime (typically called internally by xajax from get/printJavascript).
  856.      * 
  857.      * @return string 
  858.      */
  859.     function getJavascriptConfig()
  860.     {
  861.         $html  = "\t<script type=\"text/javascript\">\n";
  862.         $html .= "var xajaxRequestUri=\"".$this->sRequestURI."\";\n";
  863.         $html .= "var xajaxDebug=".($this->bDebug?"true":"false").";\n";
  864.         $html .= "var xajaxStatusMessages=".($this->bStatusMessages?"true":"false").";\n";
  865.         $html .= "var xajaxWaitCursor=".($this->bWaitCursor?"true":"false").";\n";
  866.         $html .= "var xajaxDefinedGet=".XAJAX_GET.";\n";

  1027.  
  1028.         // Add the port if needed
  1029.         if (!empty($aURL['port']) && (($aURL['scheme'] == 'http' && $aURL['port'] != 80) || ($aURL['scheme'] == 'https' && $aURL['port'] != 443))) {
