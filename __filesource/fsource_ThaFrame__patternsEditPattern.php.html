  246.     /** Transverse comma separated values into an Array **/

  275.   public function insertField($field_name, $field_data, $target, $position='after')
  276.   {
  277.     $success = false;
  278.     /** there is no easy way to insert an element into an array, so we need to
  279.     recreate it, inserting the field when we detect the $target **/
  280.     $new_fields = array();
  281.     reset($this->fields);
  282.     while (list($key, $value) = each($this->fields) ) {
  283.       if($position=='after') {
  284.         $new_fields[$key] = $value;
  285.       }
  286.       if ( $key === $target) {
  287.         if ( $field_name!='' ) {
  288.           $new_fields[$field_name] = $field_data;
  289.         } else {
  290.           $new_fields[] = $field_data;
  291.         }
  292.         $success = true;
  293.       }
  294.       if($position=='before') {
  295.         $new_fields[$key] = $value;
  296.       }
  297.     }
  298.     $this->fields = $new_fields;
  299.     
  300.     return $success;
  301.   }
