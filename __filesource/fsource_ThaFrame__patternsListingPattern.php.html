  388.   public function setQuery($sql, DbConnection $DbConnection=null, $paginate = false) {
  389.     if (is_null($DbConnection) ) {
  390.       $DbConnection = DbConnection::getInstance();
  391.     }
  392.     
  393.     if ($paginate) {
  394.       $this->_paginate = true;
  395.       
  396.       //Get a Grip of the whole thing
  397.       $sql_without_conditions = str_replace('{conditions}','',$sql);
  398.       $count_sql = "SELECT count(*) FROM ($sql_without_conditions) AS count_table;";
  399.       $total_rows = $DbConnection->getOneValue($count_sql);
  400.       
  401.       //Create some basic pattern variables
  402.       $this->_page_number = (empty($_GET['__page_number']))?'0':$_GET['__page_number'];
  403.       $this->_page_size = (empty($_GET['__page_size']))?$this->_page_size:$_GET['__page_size'];
  404.       $this->_pages = ceil($total_rows/$this->_page_size);
  405.       
  406.       if($this->_page_number > $this->_pages){
  407.         $this->_page_number = $this->_pages-1;
  408.       }
  409.       
  410.       //Reformat the query to use MySQL Limit clause
  411.       $page_start = $this->_page_number * $this->_page_size;
  412.       $sql = "$sql
  413.             LIMIT $page_start, $this->_page_size";
  414.       
  415.       $this->setPatternVariable('paginate', $this->_paginate);
  416.       $this->setPatternVariable('page_number', $this->_page_number);
  417.       $this->setPatternVariable('page_size',$this->_page_size);
  418.       $this->setPatternVariable('pages', $this->_pages);
  419.     }
  420.     
  421.     $conditions = '';
  422.     if ( count($this->_filters) ) {
  423.       
  424.       foreach($this->_filters AS $field => $filter) {
  425.         $Filter = (object)$filter;
  426.         if($Filter->type=='custom') {
  427.           //echo "Checking for filter on '$field'\n";
  428.           if( isset($_GET[$field]) ) {
  429.             $selected = stripslashes($_GET[$field]);
  430.           } else {
  431.             $selected = $this->_filters[$field]['default'];
  432.           }
  433.           //echo "Selected:$selected\n";
  434.           foreach($Filter->options AS $option){
  435.             //echo "Comparing value: {$option['value']}\n";
  436.             if ( $option['value'] == $selected) {
  437.                 //echo "Match!, condition added '{$option['condition']}'\n\n";
  438.                 $conditions .= "\nAND ";
  439.                 $conditions .= $option['condition'];
  440.                 $this->_filters[$field]['selected']= $selected;
  441.             }
  442.           }
  443.         } else if($Filter->type=='hidden') {
  444.           $conditions .= "\nAND ";
  445.           $conditions .= $Filter->condition;
  446.         }
  447.       }
  448.     }
  449.  
  450.     if ( empty($conditions) ) {
  451.       $sql = str_replace('{conditions}','',$sql);
  452.     } else {
  453.       $sql = str_replace('{conditions}', $conditions, $sql);
  454.     }
  455.     
  456.     $this->_conditions = $conditions;
  457.     $this->_sql        = $sql;
  458.     
  459.     $rows = $DbConnection->getAllRows($sql);
  460.     $this->setRows($rows);
  461.   }

  504.    * 
  505.    * @param string $config_name

  508.   public function loadConfig($config_name='default') {

  510.     
  511.     $file_name = TO_ROOT."/configs/models/{$config_name}_list.ini";
  512.     
  513.     if(!file_exists($file_name)){
  514.       Logger::log("Couldn't find config file", $file_name, LOGGER_ERROR);
  515.       return false;
  516.     }
  517.     $config = parse_ini_file($file_name, true);
  518.     
  519.     if( isset($config['__general']['page_name']) ) {
  520.       $this->setPageName($config['__general']['page_name']); 
  521.     }
  522.     unset($config['__general']);
  523.     
  524.     if( isset($config['__query']['sql']) ) {
  525.       $paginate = (isset($config['__query']['paginate']))?$config['__query']['paginate']:false;
  526.       $this->setQuery($config['__query']['sql'], DbConnection::getInstance(), $paginate); 
  527.     }
  528.     unset($config['__set_query']);
  529.     
  530.     if( isset($config['__pattern']) ) {
  531.       foreach($config['__pattern'] AS $field=>$value) {
  532.         $this->setPatternVariable($field, $value);
  533.       }
  534.     }
  535.     unset($config['__pattern']);
  536.     
  537.     if( isset($config['__commands']['hide']) ) {
  538.       foreach($config['__commands']['hide'] AS $hide) {
  539.         $this->hideField($hide);
  540.       }
  541.     }
  542.     unset($config['__commands']);
  543.     
  544.     foreach($config AS $field => $properties){
  545.       if ( strpos($field, ':')!==false ) {
  546.         list($field, $action) = explode(':', $field);
  547.         if ( $field=='__generalAction' ) {
  548.           $this->AddGeneralAction($properties['action'], $properties['title'], $properties['field'], $properties['value'], $properties['icon'], $properties['ajax']);
  549.         } else {
  550.           if($action == 'action') {
  551.             $this->addAction($field, $properties['action'], $properties['title'], $properties['icon'], $properties['ajax']);
  552.           } else if($action == 'link') {
  553.             $this->addLink($field, $properties['value'], $properties['action']);
  554.           }
  555.         }
  556.       } else {
  557.         foreach($properties AS $property => $value) {
  558.           if ($property=='format') {
  559.             $this->setFormat($field, $value);
  560.           } else if($property=='name') {
  561.             $this->setName($field, $value);
  562.           } else if($property=='class') {
  563.             $this->setClass($field, $value);
  564.           }
  565.         }
  566.       }
  567.     }
