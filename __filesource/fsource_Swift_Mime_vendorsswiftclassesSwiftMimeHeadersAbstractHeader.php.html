  360.   /**

  380.         switch($firstChar)
  381.         {
  382.           case ' ':

  388.         if (-1 == $usedLength)

  393.         
  394.         $header->setMaxLineLength(76); //Forefully override
  395.       }
  396.       else
  397.       {
  398.         $value .= $token;
  399.       }
  400.     }
  401.     
  402.     return $value;
  403.   }

  440.       }
  441.     }
  442.     if (strlen($encodedToken))
  443.     {
  444.       $tokens[] = $encodedToken;
  445.     }
  446.
